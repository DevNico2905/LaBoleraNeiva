<!DOCTYPE html>
<html lang="es">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Botón como radio con alertas</title>
  <style>
    .radio-btn {
      padding: 10px 20px;
      border: 1px solid #ccc;
      background-color: #f0f0f0;
      cursor: pointer;
      border-radius: 5px;
      transition: all 0.3s;
    }

    .radio-btn.selected {
      background-color: #4caf50;
      color: white;
      border-color: #4caf50;
    }
  </style>
</head>
<body>
  <div>
    <button class="radio-btn" data-value="Opción 1">Opción 1</button>
    <button class="radio-btn" data-value="Opción 2">Opción 2</button>
    <button class="radio-btn" data-value="Opción 3">Opción 3</button>
  </div>

  <script>
    // Seleccionamos todos los botones
    const radioButtons = document.querySelectorAll(".radio-btn");
    let selectedValue = null; // Variable para guardar la selección actual

    radioButtons.forEach((button) => {
      button.addEventListener("click", () => {
        // Removemos la clase 'selected' de todos los botones
        radioButtons.forEach((btn) => btn.classList.remove("selected"));

        // Marcamos el botón seleccionado
        button.classList.add("selected");

        // Guardamos la selección actual
        selectedValue = button.dataset.value;

        // Mostramos un alert diferente según la opción seleccionada
        if (selectedValue === "Opción 1") {
          alert("Seleccionaste la Opción 1");
        } else if (selectedValue === "Opción 2") {
          alert("Seleccionaste la Opción 2");
        } else if (selectedValue === "Opción 3") {
          alert("Seleccionaste la Opción 3");
        }
      });
    });
  </script>
</body>
</html>
